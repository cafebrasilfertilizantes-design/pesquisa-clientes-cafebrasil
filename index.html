// ================================================================ // GOOGLE APPS SCRIPT — Café Brasil Fertilizantes // Pesquisa de Clientes → Google Sheets // // INSTRUÇÕES DE CONFIGURAÇÃO: // 1. Abra o Google Sheets e crie uma planilha nova // 2. Renomeie a aba para: Respostas // 3. Clique em Extensões > Apps Script // 4. Cole TODO este código no editor // 5. Salve (Ctrl+S) // 6. Clique em Implantar > Nova implantação // 7. Tipo: App da Web // Executar como: Eu mesmo // Quem tem acesso: Qualquer pessoa // 8. Clique em Implantar e copie a URL gerada // 9. Cole a URL no arquivo HTML no campo APPS_SCRIPT_URL // ================================================================ // ID da planilha — copie da URL do seu Google Sheets // Exemplo de URL: https://docs.google.com/spreadsheets/d/ESTE_EH_O_ID/edit const SPREADSHEET_ID = 'COLE_SEU_SPREADSHEET_ID_AQUI'; const SHEET_NAME = 'Respostas'; // Cabeçalhos das colunas (ordem importa!) const HEADERS = [ 'Data/Hora', 'Nome', 'Cidade', 'Estado', 'Telefone', 'E-mail', 'Uso CICLUS', 'Como Conheceu', 'Oferta CICLUS pelo time', 'Conhece Foliares e Biológicos', 'Oferta Foliares e Biológicos pelo time', 'Utiliza Foliares e Biológicos', 'Interesse no Portfólio Foliar/Bio', 'Principal Benefício', 'Resultados Observados', 'Avaliação Suporte', 'Frequência Visitas', 'Probabilidade Ampliar', 'Dores da Lavoura (TOP 3)', 'Vendedor', 'Comentários' ]; // ---------------------------------------------------------------- // doPost — recebe os dados do formulário HTML // ---------------------------------------------------------------- function doPost(e) { try { const data = JSON.parse(e.postData.contents); saveToSheet(data); return ContentService .createTextOutput(JSON.stringify({ status: 'ok' })) .setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService .createTextOutput(JSON.stringify({ status: 'error', message: err.message })) .setMimeType(ContentService.MimeType.JSON); } } // ---------------------------------------------------------------- // doGet — responde a requisições GET (útil para teste) // ---------------------------------------------------------------- function doGet(e) { return ContentService .createTextOutput(JSON.stringify({ status: 'ok', message: 'Script ativo!' })) .setMimeType(ContentService.MimeType.JSON); } // ---------------------------------------------------------------- // saveToSheet — salva os dados na planilha // ---------------------------------------------------------------- function saveToSheet(data) { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); let sheet = ss.getSheetByName(SHEET_NAME); // Cria a aba se não existir if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); } // Cria os cabeçalhos se a planilha estiver vazia if (sheet.getLastRow() === 0) { sheet.appendRow(HEADERS); // Formata cabeçalho const headerRange = sheet.getRange(1, 1, 1, HEADERS.length); headerRange.setBackground('#2D6A2D'); headerRange.setFontColor('#FFFFFF'); headerRange.setFontWeight('bold'); headerRange.setFontSize(11); // Congela cabeçalho sheet.setFrozenRows(1); // Ajusta largura das colunas sheet.setColumnWidth(1, 160); // Data/Hora sheet.setColumnWidth(2, 200); // Nome sheet.setColumnWidth(3, 140); // Cidade sheet.setColumnWidth(4, 70); // Estado sheet.setColumnWidth(5, 150); // Telefone sheet.setColumnWidth(6, 220); // E-mail sheet.setColumnWidth(7, 280); // Uso CICLUS sheet.setColumnWidth(8, 220); // Como Conheceu sheet.setColumnWidth(9, 220); // Benefício sheet.setColumnWidth(10, 220); // Resultados sheet.setColumnWidth(11, 200); // Suporte sheet.setColumnWidth(12, 220); // Frequência sheet.setColumnWidth(13, 200); // Ampliar sheet.setColumnWidth(14, 350); // Dores sheet.setColumnWidth(15, 180); // Vendedor sheet.setColumnWidth(16, 300); // Comentários } // Monta a linha de dados na ordem dos cabeçalhos const row = [ data.timestamp || '', data.nome || '', data.cidade || '', data.estado || '', data.telefone || '', data.email || '', data.uso_ciclus || '', data.como_conheceu || '', data.oferta_ciclus || '', data.conhece_foliares || '', data.oferta_foliares || '', data.utiliza_foliares || '', data.interesse_foliar || '', data.beneficio || '', data.resultados || '', data.suporte || '', data.frequencia || '', data.ampliar || '', data.dores_ordem || '', data.vendedor || '', data.comentarios || '' ]; sheet.appendRow(row); // Aplica zebrado nas linhas const lastRow = sheet.getLastRow(); const rowRange = sheet.getRange(lastRow, 1, 1, HEADERS.length); if (lastRow % 2 === 0) { rowRange.setBackground('#f0f7f0'); } else { rowRange.setBackground('#ffffff'); } // Alinha verticalmente no topo rowRange.setVerticalAlignment('top'); rowRange.setWrap(true); } // ---------------------------------------------------------------- // setupSheet — função auxiliar para formatar a planilha manualmente // Execute esta função uma vez se quiser preparar a planilha antes // ---------------------------------------------------------------- function setupSheet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); } sheet.clearContents(); sheet.appendRow(HEADERS); const headerRange = sheet.getRange(1, 1, 1, HEADERS.length); headerRange.setBackground('#2D6A2D'); headerRange.setFontColor('#FFFFFF'); headerRange.setFontWeight('bold'); sheet.setFrozenRows(1); Logger.log('Planilha configurada com sucesso!'); }
